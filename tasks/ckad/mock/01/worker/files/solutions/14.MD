# vim 14.yaml

apiVersion: v1
kind: Pod
metadata:
  creationTimestamp: null
  labels:
    run: multi-pod
  name: multi-pod
spec:
  containers:
  - env:
    - name: type
      value: alpha
    image: nginx:alpine-slim
    name: alpha

  - env:
    - name: type
      value: beta
    image: busybox
    name: beta
    command: ["sleep","4800"]

    resources: {}
  dnsPolicy: ClusterFirst
  restartPolicy: Always
status: {}

k apply -f 14.yaml
