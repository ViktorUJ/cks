``` 
# vim 10.yaml

apiVersion: v1
kind: Pod
metadata:
  creationTimestamp: null
  labels:
    run: multi-pod
  name: multi-pod
spec:
  containers:
  - env:
    - name: name
      value: alpha
    image: nginx
    name: alpha
  - env:
    - name: name
      value: beta
    image: busybox
    name: beta 
    command: ["sleep","4800"]   
    
    resources: {}
  dnsPolicy: ClusterFirst
  restartPolicy: Always
status: {}
 
```

``` 
k apply -f 10.yaml
```
