1. Develop cilium network policy

```yml
apiVersion: "cilium.io/v2"
kind: CiliumNetworkPolicy
metadata:
  name: "allow-access-from-nginx-pod"
  namespace: "myapp"
spec:
  endpointSelector:
    matchLabels:
      run: myapp
  ingress:
  - fromEndpoints:
    - matchLabels:
        "k8s:io.kubernetes.pod.namespace": ingress-nginx
    authentication:
      mode: "required"
    toPorts:
    - ports:
      - port: "8080"
        protocol: TCP
  egress:
    - toEntities:
        - all
```
