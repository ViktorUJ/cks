```` 
# allow_prod-db_backend.yaml

apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-db-backend
  namespace: prod-db
spec:
  podSelector:
    matchLabels:
      app: mysql
  policyTypes:
    - Ingress
  ingress:
    - from:
       - podSelector:
            matchLabels:
              app: backend
         namespaceSelector:
            matchLabels:
              kubernetes.io/metadata.name: prod-stack-1

      ports:
        - protocol: TCP
          port: 80


````
